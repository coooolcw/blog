import { mapMutations } from "vuex";
import uniPopup from "@/components/uni-popup/uni-popup.vue";
import uniSubPopup from "@/components/uni-popup/uni-subpopup.vue";
import LbPicker from "@/components/lb-picker";
export default {
  data() {
    return {
      userId: null,
      contactPerson: null,
      telephone: null,
      addressDetail: null,
      tel: null,
      province: null,
      city: null,
      area: null,
      address: null,
      comProps: {
        label: "name",
        value: "code",
      },
      plist: null,
      clist: null,
      alist: null,
      pcode: null,
      ccode: null,
      acode: null,
      ptext: null,
      ctext: null,
      atext: null,
    };
  },
  components: {
    uniPopup,
    uniSubPopup,
    LbPicker,
  },
  onLoad(option) {
    console.log(JSON.parse(option.id));
    this.userId = option.id;
  },
  methods: {
    ...mapMutations(["login"]),
    close(done) {
      done();
    },
    inputChange(e) {
      const key = e.currentTarget.dataset.key;
      this[key] = e.detail.value;
    },
    navBack() {
      uni.navigateBack();
    },
    toLogin() {
      uni.redirectTo({
        url: "pages/public/auth",
      });
    },
    changeg(e) {
      console.log(e);
    },
    skipfunc() {
      // this.$refs.subpopup.open();
      uni.switchTab({
        url: "/pages/index/index",
      });
    },
    // 改变方法
    confirmp(e) {
      this.pcode = e.item.code;
      this.ptext = e.item.name;
    },
    confirmc(e) {
      this.ccode = e.item.code;
      this.ctext = e.item.name;
    },
    confirma(e) {
      this.acode = e.item.code;
      this.atext = e.item.name;
    },
    // 打开方法
    async openpro() {
      this.$refs.ppick.show();
      this.$api.load;
      let data = {
        parentCode: "100001",
      };
      var res = await this.$get("/region/region", data);
      this.plist = res.result;
      // console.log(this.plist)
    },
    async opencit() {
      if (!this.pcode) {
        return this.$api.toast("请先选择省");
      }
      this.$refs.cpick.show();
      this.$api.load;
      let data = {
        parentCode: this.pcode,
      };
      var res = await this.$get("/region/region", data);
      this.clist = res.result;
    },
    async openare() {
      if (!this.pcode || !this.ccode) {
        return this.$api.toast("请先选择市");
      }
      this.$refs.apick.show();
      this.$api.load;
      let data = {
        parentCode: this.ccode,
      };
      var res = await this.$get("/region/region", data);
      this.alist = res.result;
    },
    // 未完
    async surefunc() {
      // 这里需要调整
      if (!this.pcode || !this.ccode || !this.acode) {
        return this.$api.toast("请先选择地址");
      } else if (!this.contactPerson) {
        return this.$api.toast("请先填写收货人");
      } else if (!this.telephone) {
        return this.$api.toast("请先填写收货电话");
      }
      const params = {
        memberId: this.userId,
        addressDetail: this.addressDetail,
        area: this.acode,
        city: this.ccode,
        contactPerson: this.contactPerson,
        province: this.pcode,
        telephone: this.telephone,
        flag: true
      };
      var res = await this.$post("/auth/user/create", params);
      if (res?.code === 200) {
        this.$api.toast("注册成功");
        this.toLogin();
      }
    },
  },
};
